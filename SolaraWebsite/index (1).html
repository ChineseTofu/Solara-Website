Techniek

Code & ontwikkeling

Het hart van het enemy-systeem is een state-driven controller die verantwoordelijk is voor detectie, aanval-timing en gedragsovergangen. De EnemyController beheert attack-cooldowns, telegraphing via indicators en state switching tussen Chase en Attack.

Dankzij coroutines wordt het volledige aanvalspatroon (wind-up → indicator → execute → reset) strak gesynchroniseerd zonder complexe state machines. Dit maakt het systeem zowel uitbreidbaar als makkelijk te balancen via ScriptableObjects.

C# Unity 2D Coroutines State Machine Component-architectuur 
using System;


public class EnemyController : MonoBehaviour
{
    public Rigidbody2D rb { get; private set; }
    public EnemyMovement enemyMovement { get; private set; }
    public GameObject player { get; private set; }
    public EnemyState enemyState { get; private set; }
    public PlayerDetector playerDetector { get; private set; }
    public BaseEnemyStats BaseEnemyStats => baseEnemyStats;
    public Vector2 targetPositionLock { get; private set; }

    [SerializeField] private BaseEnemyStats baseEnemyStats;

    private EnemyHealthComponent enemyHealthComponent;
    private AttackIndicator attackIndicator;
    private IAttack iAttack;

    private bool isAttacking = false;
    private float lastAttackTime;

    private void Awake()
    {
        enemyMovement = GetComponent<EnemyMovement>();
        enemyHealthComponent = GetComponent<EnemyHealthComponent>();
        enemyHealthComponent.Initialize(baseEnemyStats.HP);

        rb = GetComponent<Rigidbody2D>();
        playerDetector = GetComponent<PlayerDetector>();
        iAttack = GetComponent<IAttack>();
        attackIndicator = GetComponent<AttackIndicator>();
    }

    public void CheckPlayerInAttackRange()
    {
        if (isAttacking) return;
        if (player == null) return;

        float distance = Vector2.Distance(transform.position, player.transform.position);

        if (distance <= baseEnemyStats.AtkRange)
        {
            StartCoroutine(AttackRoutine());
        }
    }

    private IEnumerator AttackRoutine()
    {
        if (Time.time < lastAttackTime + BaseEnemyStats.AtkCooldown)
            yield break;

        lastAttackTime = Time.time;
        isAttacking = true;
        enemyState = EnemyState.Attack;

        targetPositionLock = player.transform.position;

        if (attackIndicator != null)
            attackIndicator.DrawAttackIndicator();

        yield return new WaitForSeconds(baseEnemyStats.WindUpTime);

        if (attackIndicator != null)
            attackIndicator.HideIndicator();

        yield return iAttack.Attack();

        isAttacking = false;
        enemyState = EnemyState.Chase;
    }

    private void HandleDeath()
    {
        QuestManager.Instance.OnEnemyKilled(baseEnemyStats.GetEnemyType);
        Destroy(gameObject);
    }

    public enum EnemyState
    {
        Chase,
        Attack,
    }
}
